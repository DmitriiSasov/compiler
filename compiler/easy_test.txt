
fun sort(arr:Array<Int>, isIncrease:Boolean):Unit {
    var i:Int = 0
    var arrSize:Int = arr.size
    while(i < arrSize){
        var j:Int = i
        var currentMinIndex:Int = j
        while (j < arrSize){
            if (isIncrease){
                if (arr[j] < arr[currentMinIndex]) {
                    currentMinIndex = j
                }
            } else {
                if (arr[j] > arr[currentMinIndex]) {
                    currentMinIndex = j
                }
            }            
            j = j + 1
        }
        var tmp:Int = arr[i]
        arr[i] = arr[currentMinIndex]
        arr[currentMinIndex] = tmp
        i = i + 1
    }
}


fun main():Unit{
    
    var arr:Array<Array<Int>> = arrayOf(arrayOf(4,5,1,3,2), arrayOf(4,5,1,3,7), arrayOf(4,5,1,8,9))
    print(arr)
    var arrSize:Int = arr.size
    var i:Int = 0
    while (i < arrSize){
        sort(arr[i], false)
        i = i + 1
    }
    print(arr)
}
